🚀 Dual Deployment System

🖥️ CLI Deployment

# One-script deployment
curl -sSL https://dirk-brain.com/install | bash

# Quick deploy to current project
./deploy-dirk-brain.sh

# Advanced deployment
dirk deploy --agents claude,gemini,sub-agents \
--hooks git,file-watch,godmode \
--project /path/to/project \
--template react-enterprise

# GOD Mode deployment
godmode deploy "Full-stack React app with auth" \
--parallel --auto-recovery
